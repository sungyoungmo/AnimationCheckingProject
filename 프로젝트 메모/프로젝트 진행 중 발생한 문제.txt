public void AttackCheck()
{
    if (!photonView.IsMine) return;

    foreach (var keyInput in skillKeys)
    {
        if (!_isAttack && Input.GetKeyDown(keyInput))
        {
            playerController.TransitionToState(playerController.skillState,keyInput);
            break;
        }
    }
}
